def initialize(command, observer, time)
  @command = command
  @task = Concurrent::Future.new {
    timeout(time){
      out_r, out_w = IO.pipe
      err_r, err_w = IO.pipe
      spawn @command, {out: out_w, err: err_w}
      out_w.close
      err_w.close
      out_r.read
    }
  }
  @task.add_observer(observer)
end
